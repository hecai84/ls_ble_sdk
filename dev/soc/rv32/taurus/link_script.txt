MEMORY
{
    FLASH (rx) : ORIGIN = 0x800000, LENGTH = 0x80000
    RAM (rwx) : ORIGIN = 0x400000, LENGTH = 0xd000
}
ENTRY(Reset_Handler)
SECTIONS
{
	.text :
	{
        *(.text.Reset_Handler)
		KEEP(*(.init))
		KEEP(*(.fini))

		*crtbegin.o(.ctors)
		*crtbegin?.o(.ctors)
		*(EXCLUDE_FILE(*crtend?.o *crtend.o) .ctors)
		*(SORT(.ctors.*))
		*(.ctors)


		*crtbegin.o(.dtors)
		*crtbegin?.o(.dtors)
		*(EXCLUDE_FILE(*crtend?.o *crtend.o) .dtors)
		*(SORT(.dtors.*))
		*(.dtors)

		*(.eh_frame*)
		*(.text* .srodata* .rodata*)
		. = ALIGN(4);
	}>FLASH
    __data_lma__ = .;
	.data :
	{
		__data_start__ = .;
        *(.sdata*)
		*(.data*)
		*(.xip_banned)
		*(vtable)
		__data_end__ = ALIGN(4);
        _edata = __data_end__;
	}>RAM AT>FLASH

	.bss (NOLOAD):
	{
		__bss_start__ = .;
        *(.sbss*)
		*(.bss*)
		*(SECTION_RTT)
		__bss_end__ = ALIGN(4);
        PROVIDE(_end = .);
		PROVIDE(end = .);
	}>RAM

	__StackTop = ORIGIN(RAM) + LENGTH(RAM);
	PROVIDE(__global_pointer$ = __data_start__ + (4K / 2) );	
}
